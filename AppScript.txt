const GEMINI_API_KEY = PropertiesService.getScriptProperties().getProperty("GEMINI_API_KEY");


function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getActiveSheet();
   
    var timestamp = new Date();
    var vibration = data.vibration == 1 ? "🫨 VIBRATION!" : "STABLE";
    var event = data.event || "NORMAL";
    var people = data.people || 0;
   
    sheet.appendRow([timestamp, data.device, data.distance, vibration, event, people]);
   
    if (event === "DAILY RESET") {
      analyzeDataWithGemini(sheet);
    }
   
    return ContentService.createTextOutput(JSON.stringify({"result":"success"})).setMimeType(ContentService.MimeType.JSON);
  } catch (f) {
    return ContentService.createTextOutput(JSON.stringify({"result":"error", "error": f.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}


function analyzeDataWithGemini(sheet) {
  try {
    var lastRow = sheet.getLastRow();
    var numRows = Math.min(lastRow - 1, 100);
    if (numRows < 1) return;


    var dataRange = sheet.getRange(lastRow - numRows + 1, 1, numRows, 6);
    var values = dataRange.getValues();
   
    var maxPeople = 0;
    var alarmCount = 0;
    for (var i = 0; i < values.length; i++) {
      if (values[i][5] > maxPeople) maxPeople = values[i][5];
      if (values[i][4].toString().includes("ALARM")) alarmCount++;
    }


    var prompt = "สรุปภาพรวมความปลอดภัยจาก Log นี้ เป็นภาษาไทย ห้ามเกิน 25 คำ: " + values.slice(-10).join("\n");
    var url = "https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=" + GEMINI_API_KEY;
    var response = UrlFetchApp.fetch(url, { "method": "post", "contentType": "application/json", "payload": JSON.stringify({ "contents": [{ "parts": [{ "text": prompt }] }] }) });
    var aiSummary = JSON.parse(response.getContentText()).candidates[0].content.parts[0].text.trim();



   
    var headerRange = sheet.getRange("J1:M1");
    var dataRangeBox = sheet.getRange("J2:M2");
   
    headerRange.setValues([["📅 อัปเดตล่าสุด", "🤖 AI Summary (<25 คำ)", "👥 ยอดคนรวม", "🚨 จำนวน Alarm"]]);
    dataRangeBox.setValues([[new Date(), aiSummary, maxPeople, alarmCount]]);


    headerRange.setBackground("#1a202c")
               .setFontColor("#ffffff")
               .setFontWeight("bold")
               .setHorizontalAlignment("center")
               .setVerticalAlignment("middle");


    dataRangeBox.setBackground("#f7fafc")
                .setFontColor("#2d3748")
                .setHorizontalAlignment("center")
                .setVerticalAlignment("middle")
                .setBorder(true, true, true, true, true, true, "#cbd5e0", SpreadsheetApp.BorderStyle.SOLID);


    sheet.getRange("K2").setWrap(true);

    sheet.setColumnWidth(10, 160); // J: เวลา
    sheet.setColumnWidth(11, 350); // K: AI Summary (กว้างหน่อยเพื่อให้ชัด)
    sheet.setColumnWidth(12, 100); // L: ยอดคน
    sheet.setColumnWidth(13, 120); // M: Alarm

    sheet.setFrozenRows(2);


  } catch (error) {
    sheet.getRange("K2").setValue("⚠️ AI Error: " + error.toString()).setFontColor("red");
  }
}
